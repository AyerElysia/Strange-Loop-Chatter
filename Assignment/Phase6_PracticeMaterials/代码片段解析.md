# plugins/life_engine/snn/core.py::LIFNeuronGroup

class LIFNeuronGroup:
    """LIF（Leaky Integrate-and-Fire）神经元群。"""

    def __init__(
        self,
        n_neurons: int,
        tau_m: float = 20.0,     # 膜电位时间常数（ms）
        threshold: float = 1.0,   # 脉冲阈值
        reset_v: float = 0.0,     # 重置电压
        tau_trace: float = 25.0,  # Trace时间常数（ms）
    ):
        self._n = n_neurons
        self._tau_m = tau_m
        self._threshold = threshold
        self._reset_v = reset_v
        self._tau_trace = tau_trace

        # 膜电位（每个神经元一个V）
        self._voltages = np.zeros(n_neurons, dtype=np.float64)

        # Trace（记录脉冲历史）
        self._traces = np.zeros(n_neurons, dtype=np.float64)

    def step(self, input_current: np.ndarray) -> np.ndarray:
        """神经元步进更新（LIF动力学）。"""
        # 1. 膜电位更新（Leaky Integrate）
        dV = (-self._voltages + input_current) / self._tau_m
        self._voltages += dV * dt  # dt=1.0ms

        # 2. 脉冲产生（Fire）
        spikes = (self._voltages > self._threshold).astype(np.float64)

        # 3. 重置膜电位（spike后重置）
        self._voltages[spikes > 0] = self._reset_v

        # 4. Trace更新（记录脉冲历史）
        d_trace = (-self._traces + spikes) / self._tau_trace
        self._traces += d_trace * dt

        return spikes
```

     tau_m=20ms：膜电位时间常数（生物典型值）

   - self._voltages += dV * dt
     Euler方法更新膜电位

2. 脉冲产生：
   - spikes = (self._voltages > self._threshold).astype(np.float64)
     阈值检测：膜电位超过阈值→产生脉冲

     Trace衰减+脉冲增加
     tau_trace=25ms：Trace时间常数

   - self._traces += d_trace * dt
     Trace记录脉冲历史（用于STDP学习）

4. 返回脉冲：
   - return spikes
     返回脉冲信号（用于下一层输入）

系统级意义：
- 模拟生物LIF神经元动力学

# plugins/life_engine/snn/core.py::STDPSynapse

class STDPSynapse:
    """STDP（Spike-Timing-Dependent Plasticity）突触学习。"""

    def __init__(
        self,
        n_pre: int,
        n_post: int,
        lr_plus: float = 0.01,  # LTP学习率
        lr_minus: float = 0.01,  # LTD学习率
        w_min: float = 0.0,      # 权重下界
        w_max: float = 1.0,      # 权重上界
    ):
        self._lr_plus = lr_plus
        self._lr_minus = lr_minus
        self._w_min = w_min
        self._w_max = w_max

        # 突触权重矩阵（n_pre × n_post）
        self._W = np.random.uniform(0.4, 0.6, size=(n_pre, n_post))

        # Trace变量（记录脉冲历史）
        self.trace_pre = np.zeros(n_pre, dtype=np.float64)
        self.trace_post = np.zeros(n_post, dtype=np.float64)
        self.trace_decay = 0.95  # Trace衰减因子

    def update_soft(
        self,
        pre_activity: np.ndarray,
        post_activity: np.ndarray,
        reward: float = 0.0,
    ):
        """软STDP更新：基于连续活跃度的学习。"""
        # 1. Trace更新（记录脉冲历史）
        self.trace_pre = self.trace_pre * self.trace_decay + pre_activity
        self.trace_post = self.trace_post * self.trace_decay + post_activity

        # 2. Reward调制
        reward_factor = 1.0 + np.clip(reward, -1.0, 1.0)

        # 3. LTP（长时程增强）
        if np.sum(post_activity) > 0.05:  # 后神经元活跃
            dw_plus = self._lr_plus * np.outer(post_activity, self.trace_pre)
            self._W += dw_plus * max(reward_factor, 0.1)  # 正reward强化LTP

        # 4. LTD（长时程抑制）
        if np.sum(pre_activity) > 0.05:  # 前神经元活跃
            dw_minus = -self._lr_minus * np.outer(self.trace_post, pre_activity)
            self._W += dw_minus * max(2.0 - reward_factor, 0.1)  # 负reward强化LTD

        # 5. 权重约束
        np.clip(self._W, self._w_min, self._w_max, out=self._W)
```

     避免初始权重极端值

2. Trace更新：

   - self.trace_post = ...

3. Reward调制：

系统级意义：

def step(self, input_vec: np.ndarray, reward: float = 0.0) -> np.ndarray:
    """真实输入步：完整的SNN计算+学习流程。"""
    # 1. 输入特征缩放
    input_scaled = input_vec * self._input_gain

    # 2. 隐藏层激活
    current_hidden = self.syn_in_hid.forward(input_scaled)
    spikes_hidden = self.hidden.step(current_hidden)

    # 3. 输出层激活
    hidden_spike_signal = spikes_hidden.astype(np.float64) * self._hidden_spike_gain
    current_output = self.syn_hid_out.forward(hidden_spike_signal)
    spikes_output = self.output.step(current_output)

    # 4. STDP学习（权重调整）
    hidden_v = self.hidden.get_state()
    output_v = self.output.get_state()

    # 软STDP（基于膜电位的连续活跃度）
    soft_hidden = _sigmoid(hidden_v, center=self.hidden.threshold * 0.5)
    soft_output = _sigmoid(output_v, center=self.output.threshold * 0.5)
    input_activity = np.abs(input_vec) / max(np.max(np.abs(input_vec)), 0.01)

    # 突触权重更新（输入→隐藏）
    self.syn_in_hid.update_soft(input_activity, soft_hidden, reward=reward)

    # 突触权重更新（隐藏→输出）
    self.syn_hid_out.update_soft(soft_hidden, soft_output, reward=reward)

    # 5. 驱动输出（EMA平滑）
    raw_output = self.output.get_state()
    self._output_ema = (1 - self._ema_alpha) * self._output_ema + self._ema_alpha * raw_output

    self._real_step_count += 1
    self.tick_count += 1

    return self._output_ema.copy()

# plugins/life_engine/neuromod/engine.py::Modulator

@dataclass
class Modulator:
    """单个神经调质：影响情感状态和行为倾向。"""

    name: str
    value: float = 0.5          # 当前浓度 [0, 1]
    baseline: float = 0.5       # 基线浓度（稳态目标）
    tau: float = 1800.0         # 时间常数（秒）

    def update(self, stimulus: float, dt: float):
        """ODE更新：向基线回归 + 外部刺激。"""
        # 1. 衰减（向基线回归）
        decay_rate = 1.0 / self.tau
        decay = decay_rate * (self.baseline - self.value) * dt

        # 2. 刺激增加（边际效应递减）
        headroom = 1.0 - abs(self.value - 0.5) * 2.0
        impulse = stimulus * max(headroom, 0.1) * (dt / self.tau) * 10.0

        # 3. 更新浓度
        self.value += decay + impulse
        self.value = max(0.0, min(1.0, self.value))

    def set_baseline(self, new_baseline: float):
        """更新基线浓度（长期适应）。"""
        self.baseline = max(0.0, min(1.0, new_baseline))
```

```
关键机制解析：

1. 衰减机制（向基线回归）：
   - decay_rate = 1.0 / self.tau
     衰减速率：tau=1800秒（30分钟）

   - decay = decay_rate * (self.baseline - self.value) * dt
     衰减量：向基线回归（稳态调节）
     - value > baseline → decay<0（下降）
     - value < baseline → decay>0（上升）

2. 刺激增加（边际效应递减）：
   - headroom = 1.0 - abs(self.value - 0.5) * 2.0
     边际空间：离极端值越远越容易变化
     - value=0.5 → headroom=1.0（容易变化）
     - value=0.9 → headroom=0.2（难变化）

   - impulse = stimulus * max(headroom, 0.1) * (dt/tau) * 10.0
     刺激增加：stimulus × headroom × 时间因子

3. 浓度更新：
   - self.value += decay + impulse
     衰减 + 刺激增加

   - self.value = max(0.0, min(1.0, self.value))
     约束到[0, 1]范围

4. 基线调整：
   - self.baseline = max(0.0, min(1.0, new_baseline))
     长期基线调整（性格特质形成）

系统级意义：
- 调质层衰减回归（情感自然波动）

# plugins/life_engine/neuromod/engine.py::ModulatorSystem

class ModulatorSystem:
    """调质层集合：5维神经调质系统。"""

    def __init__(self):
        self.modulators = {
            "curiosity": Modulator(name="curiosity", tau=1800.0),
            "sociability": Modulator(name="sociability", tau=1800.0),
            "diligence": Modulator(name="diligence", tau=1800.0),
            "contentment": Modulator(name="contentment", tau=1800.0),
            "energy": Modulator(name="energy", tau=1800.0),
        }

    def tick(self, stimuli: dict[str, float], dt: float = 30.0):
        """系统tick：更新所有调质浓度。"""
        for name, modulator in self.modulators.items():
            stimulus = stimuli.get(name, 0.0)
            modulator.update(stimulus, dt)

async def reinforce_coactivated(
    db: sqlite3.Connection,
    node_ids: list[str],
    learning_rate: float = 0.1,
):
    """Hebbian学习：强化共激活节点之间的连接。"""
    if len(node_ids) < 2:
        return

    # 1. 获取节点激活强度
    activations = {}
    for node_id in node_ids:
        cursor = db.execute(
            "SELECT activation_strength FROM memory_nodes WHERE node_id = ?",
            (node_id,)
        )
        row = cursor.fetchone()
        if row:
            activations[node_id] = float(row[0])

    # 2. 遍历所有共激活节点对
    for i, source_id in enumerate(node_ids):
        for target_id in node_ids[i+1:]:
            # 3. 计算共激活强度
            source_activation = activations.get(source_id, 0.5)
            target_activation = activations.get(target_id, 0.5)
            coactivation_strength = source_activation * target_activation

            # 4. 强化连接
            await create_or_update_edge(
                db,
                source_id=source_id,
                target_id=target_id,
                edge_type="ASSOCIATES",
                reason="Hebbian共激活强化",
                strength_change=learning_rate * coactivation_strength,
            )
```

# plugins/life_engine/memory/search.py::spread_activation

async def spread_activation(
    seed_node_ids: list[str],
    max_steps: int = 3,
    activation_threshold: float = 0.3,
) -> list[tuple[str, float]]:
    """激活扩散联想：主动联想相关记忆。"""
    # 1. 初始激活（种子节点）
    activated = {}
    for node_id in seed_node_ids:
        activated[node_id] = 1.0  # 激活强度=1.0

    # 2. 扩散步骤
    for step in range(max_steps):
        new_activated = {}

        # 从已激活节点扩散
        for source_id, source_strength in activated.items():
            # 查找连接节点
            edges = await get_edges_from(source_id)

            for edge in edges:
                target_id = edge.target_id
                edge_weight = edge.weight

                # 3. 计算扩散激活
                spread_strength = source_strength * edge_weight * 0.7

                if spread_strength > activation_threshold:
                    if target_id not in activated:
                        new_activated[target_id] = spread_strength

        # 合并新激活
        for node_id, strength in new_activated.items():
            if node_id not in activated:
                activated[node_id] = strength

    # 4. 返回激活节点（按强度排序）
    return sorted(activated.items(), key=lambda x: x[1], reverse=True)
```

class HabitTracker:
    """习惯追踪：记录长期行为模式。"""

    def __init__(self):
        self._habits: dict[str, Habit] = {}

    def record_habit(self, habit_name: str):
        """记录习惯行为。"""
        habit = self._habits.get(habit_name, Habit(name=habit_name))

        # 增加计数
        habit.total_count += 1

        # 计算连续天数（streak）
        today = datetime.now().date()
        if habit.last_date == today - timedelta(days=1):
            # 连续 → streak增加
            habit.streak += 1
        elif habit.last_date == today:
            # 今天已记录 → 不增加streak
            pass
        else:
            # 断开 → streak重置
            habit.streak = 1

        habit.last_date = today

        # 计算习惯强度
        habit.strength = self._compute_strength(habit)

        self._habits[habit_name] = habit

    def _compute_strength(self, habit: Habit) -> float:
        """计算习惯强度。"""
        # Streak权重（60%）
        streak_score = min(habit.streak / 14.0, 1.0)  # 14天=满分

        # 总次数权重（40%）
        count_score = min(habit.total_count / 50.0, 1.0)  # 50次=满分

        return 0.6 * streak_score + 0.4 * count_score
